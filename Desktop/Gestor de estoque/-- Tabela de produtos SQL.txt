-- Tabela de produtos
CREATE TABLE Produtos (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100),
    codigo_barras VARCHAR(50),
    volume_ml INT,
    descricao VARCHAR(255)
);

-- Tabela de estoque
CREATE TABLE Estoque (
    id INT PRIMARY KEY AUTO_INCREMENT,
    id_produto INT,
    data_entrada DATE,
    data_validade DATE,
    quantidade INT,
    localizacao VARCHAR(100),
    FOREIGN KEY (id_produto) REFERENCES Produtos(id)
);
